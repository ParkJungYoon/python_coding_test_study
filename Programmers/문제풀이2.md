#### [소수 만들기](https://programmers.co.kr/learn/courses/30/lessons/12977)

```python
from itertools import combinations

def solution(nums):
    combi = list(combinations(nums,3))
    count = 0
    for i in combi:
        prime = isPrime(sum(i))
        if prime:
            count += 1
    return count

def isPrime(num):
    if num == 1:
        return False
    for i in range(2, int(num**(1/2))+1):
        if num % i == 0:
            return False
    return True
```
